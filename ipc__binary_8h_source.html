165 struct MessageSequence {
166 auto operator()(ConnectionData<Proto, Read, Write, Alloc>* data) const {
167 return repeat(let_value_with(
168 [data](Array<byte, sizeof(MessageHeader)>& buffer) {
169 return async_read_exactly(data->read, buffer.span()) | let_value([data, &buffer] {
170 auto header = util::bit_cast<MessageHeader>(buffer);
171 auto const total_size = u32(header.message_size);
172 return just_from([data, total_size] {
173 return data->receive_buffer.reserve(total_size);
174 }) |
175 let_value([data, header] {
176 return async_read_exactly(data->read, Span { data->receive_buffer.data(),
177 header.message_size.value() -
178 sizeof(MessageHeader) }) |
179 let_value([data, header] {
180 return message_decode<Proto, ClientOrServer>(
181 header, Span { data->receive_buffer.data(),
182 header.message_size.value() });
183 });
184 });
185 });
186 },
187 make_deferred<Array<byte, sizeof(MessageHeader)>>()));
188 }
